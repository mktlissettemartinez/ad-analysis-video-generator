function syncSelectedObject(){const obj=currentObject();if(!obj)return;const nodes=ui.objectLayerHost.querySelectorAll('.object-layer');let node=null;nodes.forEach(function(n){if(n.dataset.id===obj.id)node=n;});if(node)applyObjectToNode(node,obj,state.slides[state.currentSlide]);}
async function recolorObjectDataUrl(dataUrl,hex){const target=hexToRgb(hex);if(!target)return dataUrl;const img=await loadImage(dataUrl),c=canvasFromImage(img),ctx=c.getContext('2d',{willReadFrequently:true}),d=ctx.getImageData(0,0,c.width,c.height),thsl=rgbToHsl(target[0],target[1],target[2]);for(let i=0;i<d.data.length;i+=4){if(d.data[i+3]<8)continue;const hsl=rgbToHsl(d.data[i],d.data[i+1],d.data[i+2]),sat=Math.max(thsl[1]*.88,hsl[1]*.18),light=clamp(hsl[2]*.88+thsl[2]*.12,0,1),rgb=hslToRgb(thsl[0],sat,light);d.data[i]=rgb[0];d.data[i+1]=rgb[1];d.data[i+2]=rgb[2];}ctx.putImageData(d,0,0);return c.toDataURL('image/png');}
async function applySelectedObjectColor(){const obj=currentObject();if(!obj)return;ui.applyObjectColor.disabled=true;try{obj.dataUrl=await recolorObjectDataUrl(obj.originalDataUrl,ui.objectColor.value);syncSelectedObject();}finally{ui.applyObjectColor.disabled=false;}}
function resetSelectedObjectColor(){const obj=currentObject();if(!obj)return;obj.dataUrl=obj.originalDataUrl;syncSelectedObject();}
function deleteSelectedObject(){const slide=state.slides[state.currentSlide],obj=currentObject();if(!slide||!obj)return;if(!confirm('¿Eliminar este elemento separado?'))return;slide.objects=slide.objects.filter(function(o){return o.id!==obj.id;});state.selectedObjectId=null;renderCurrentSlide();updateSummary();}
