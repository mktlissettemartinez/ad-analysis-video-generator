function renderCurrentSlide(){const slide=state.slides[state.currentSlide];if(!slide)return;ensureSlideObjects(slide);ui.slideBackground.src=slide.backgroundDataUrl;fitStage();ui.objectLayerHost.innerHTML='';ui.textLayerHost.innerHTML='';slide.objects.forEach(function(obj){ui.objectLayerHost.appendChild(createObjectLayerElement(obj,slide));});slide.lines.forEach(function(line){ui.textLayerHost.appendChild(createTextLayerElement(line,slide));});ui.selectionCanvas.width=slide.width;ui.selectionCanvas.height=slide.height;if(!state.selection.active)ui.selectionCanvas.getContext('2d').clearRect(0,0,slide.width,slide.height);ui.noSelection.classList.remove('hidden');ui.textProperties.classList.add('hidden');hideObjectProperties();}
function selectText(id){state.selectedObjectId=null;hideObjectProperties();if(ui.objectLayerHost)ui.objectLayerHost.querySelectorAll('.object-layer').forEach(function(n){n.classList.remove('selected');});state.selectedTextId=id;ui.textLayerHost.querySelectorAll('.text-layer').forEach(function(n){n.classList.toggle('selected',n.dataset.id===id);});const line=currentLine();if(!line)return;ui.noSelection.classList.add('hidden');ui.textProperties.classList.remove('hidden');ui.textContent.value=line.text;ui.textSize.value=line.fontSize.toFixed(1);ui.textColor.value=line.color;ui.textAlign.value=line.align;ui.textBold.checked=line.bold;}
function selectSlide(index){saveActivePropertyValues();cancelSelectionMode();state.currentSlide=clamp(index,0,state.slides.length-1);state.selectedTextId=null;state.selectedObjectId=null;renderThumbs();renderCurrentSlide();}
function resetCurrentSlide(){const slide=state.slides[state.currentSlide];if(!slide)return;if(!confirm('¿Restaurar esta diapositiva y eliminar las capas separadas manualmente?'))return;slide.lines=JSON.parse(JSON.stringify(slide.originalLines));slide.objects=[];slide.backgroundDataUrl=slide.baseBackgroundDataUrl||slide.backgroundDataUrl;state.selectedTextId=null;state.selectedObjectId=null;renderCurrentSlide();renderThumbs();updateSummary();}
function updateSummary(){ui.slideCount.textContent=state.slides.length;ui.textCount.textContent=state.slides.reduce(function(s,x){return s+x.lines.length;},0);const objectTotal=state.slides.reduce(function(s,x){return s+(x.objects?x.objects.length:0);},0),summary=ui.editorView.querySelector('.summary');let badge=$('objectCountHybrid');if(!badge){const span=document.createElement('span');span.innerHTML='<b id="objectCountHybrid">0</b> elementos separados';summary.appendChild(span);badge=$('objectCountHybrid');}badge.textContent=objectTotal;}
