function extractLinesFromOcr(data,scale,width,height){let raw=Array.isArray(data.lines)?data.lines:[];if(!raw.length&&Array.isArray(data.blocks))raw=data.blocks.flatMap(function(b){return (b.paragraphs||[]).flatMap(function(p){return p.lines||[];});});if(!raw.length&&Array.isArray(data.words))raw=groupWords(data.words);const lines=raw.map(function(line){const b=line.bbox||{};return{text:String(line.text||'').replace(/\s+/g,' ').trim(),confidence:Number(line.confidence==null?line.conf:line.confidence)||0,bbox:{x0:clamp((Number(b.x0)||0)/scale,0,width),y0:clamp((Number(b.y0)||0)/scale,0,height),x1:clamp((Number(b.x1)||1)/scale,1,width),y1:clamp((Number(b.y1)||1)/scale,1,height)}};}).filter(function(line){if(!isReliableLine(line))return false;const lower=line.text.toLowerCase();if(/notebook|a bool|an notebook/.test(lower))return false;if(line.bbox.y0>height*.82&&line.bbox.x0>width*.55)return false;return true;});lines.sort(function(a,b){return a.bbox.y0-b.bbox.y0||a.bbox.x0-b.bbox.x0;});const result=[];lines.forEach(function(line){const duplicate=result.findIndex(function(r){return overlapRatio(r.bbox,line.bbox)>.72;});if(duplicate>=0){if(line.confidence>result[duplicate].confidence)result[duplicate]=line;}else result.push(line);});return result;}
function isReliableLine(line){const text=line.text||'',compact=text.replace(/\s/g,'');if(!compact||line.confidence<68||compact.length<3)return false;const letters=(compact.match(/[A-Za-zÁÉÍÓÚÜÑáéíóúüñ]/g)||[]).length,accepted=(compact.match(/[A-Za-zÁÉÍÓÚÜÑáéíóúüñ0-9.,;:¿?¡!()/%+\-–—'“”"&]/g)||[]).length;if(letters<2||accepted/compact.length<.88)return false;if(/([|\\/<>_=~^]){2,}/.test(text))return false;if(/\b(?:NULL|MAX)\b/i.test(text)&&text.split(/\s+/).length<=2)return false;return true;}
