function buildConnectedSelection(imageCanvas,posCanvas,negCanvas,tolerance){const maxW=820,scale=Math.min(1,maxW/imageCanvas.width),w=Math.max(1,Math.round(imageCanvas.width*scale)),h=Math.max(1,Math.round(imageCanvas.height*scale));const imgC=createMaskCanvas(w,h);imgC.getContext('2d').drawImage(imageCanvas,0,0,w,h);const data=imgC.getContext('2d',{willReadFrequently:true}).getImageData(0,0,w,h).data;const pos=resizeMaskCanvas(posCanvas,w,h).getContext('2d').getImageData(0,0,w,h).data,neg=resizeMaskCanvas(negCanvas,w,h).getContext('2d').getImageData(0,0,w,h).data,seeds=[],palette=[];let seen=0;for(let i=0;i<w*h;i++){if(pos[i*4+3]>20){seeds.push(i);if(seen%45===0&&palette.length<64)palette.push([data[i*4],data[i*4+1],data[i*4+2]]);seen++;}}if(!seeds.length)throw new Error('Pinta sobre el elemento antes de crear la capa.');if(!palette.length){const i=seeds[0];palette.push([data[i*4],data[i*4+1],data[i*4+2]]);}const candidate=new Uint8Array(w*h),threshold=Math.max(18,tolerance)*2.15;for(let i=0;i<w*h;i++){if(neg[i*4+3]>20)continue;const r=data[i*4],g=data[i*4+1],b=data[i*4+2];let best=1e9;for(let p=0;p<palette.length;p++){const c=palette[p],d=Math.sqrt((r-c[0])**2+(g-c[1])**2+(b-c[2])**2);if(d<best)best=d;if(best<=threshold)break;}if(best<=threshold||pos[i*4+3]>20)candidate[i]=1;}const connected=new Uint8Array(w*h),queue=new Int32Array(w*h);let q0=0,q1=0;seeds.forEach(function(s){if(candidate[s]&&!connected[s]){connected[s]=1;queue[q1++]=s;}});while(q0<q1){const idx=queue[q0++],x=idx%w,y=(idx/w)|0;for(let yy=-1;yy<=1;yy++)for(let xx=-1;xx<=1;xx++){if(!xx&&!yy)continue;const nx=x+xx,ny=y+yy;if(nx<0||ny<0||nx>=w||ny>=h)continue;const ni=ny*w+nx;if(candidate[ni]&&!connected[ni]){connected[ni]=1;queue[q1++]=ni;}}}let mask=binaryDilate(connected,w,h,1);mask=binaryErode(mask,w,h,1);mask=binaryDilate(mask,w,h,1);let count=0;for(let i=0;i<mask.length;i++)if(mask[i])count++;if(count<40)throw new Error('No se pudo aislar un área suficiente. Pinta más partes del elemento o aumenta la precisión.');const maskCanvas=createMaskCanvas(w,h),md=maskCanvas.getContext('2d').createImageData(w,h);for(let i=0;i<mask.length;i++)if(mask[i])md.data[i*4+3]=255;maskCanvas.getContext('2d').putImageData(md,0,0);return{maskCanvas:maskCanvas,scale:scale,w:w,h:h,count:count};}
async function recolorCurrentBackground(){const slide=state.slides[state.currentSlide];if(!slide)return;const target=hexToRgb(ui.primaryColor.value);if(!target){alert('Escribe un color hexadecimal válido.');return;}const img=await loadImage(slide.backgroundDataUrl),canvas=canvasFromImage(img),ctx=canvas.getContext('2d',{willReadFrequently:true}),pixels=ctx.getImageData(0,0,canvas.width,canvas.height),thsl=rgbToHsl(target[0],target[1],target[2]);for(let i=0;i<pixels.data.length;i+=4){const hsl=rgbToHsl(pixels.data[i],pixels.data[i+1],pixels.data[i+2]),isWarm=hsl[0]<.16||hsl[0]>.92,isHighlyColored=hsl[1]>.55&&hsl[2]>.28;if(isWarm&&isHighlyColored)continue;const strength=clamp((.78-hsl[2])*1.2+(hsl[1]<.22?.22:0),.25,.92),newH=hsl[0]*(1-strength)+thsl[0]*strength,newS=clamp(hsl[1]*(1-strength*.55)+thsl[1]*strength*.9,0,1),rgb=hslToRgb(newH,newS,hsl[2]);pixels.data[i]=rgb[0];pixels.data[i+1]=rgb[1];pixels.data[i+2]=rgb[2];}ctx.putImageData(pixels,0,0);slide.backgroundDataUrl=canvas.toDataURL('image/jpeg',.96);const thumb=createMaskCanvas(320,Math.round(320*slide.height/slide.width));thumb.getContext('2d').drawImage(canvas,0,0,thumb.width,thumb.height);slide.thumbnailDataUrl=thumb.toDataURL('image/jpeg',.82);renderThumbs();renderCurrentSlide();}
function installBackgroundButton(){if($('recolorBackgroundBtn'))return;const button=document.createElement('button');button.id='recolorBackgroundBtn';button.className='layer-button';button.textContent='Cambiar fondo actual';button.addEventListener('click',recolorCurrentBackground);ui.isolateElementBtn.parentElement.insertBefore(button,ui.isolateElementBtn.nextSibling);}
installBackgroundButton();
