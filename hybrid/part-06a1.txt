function currentObject(){const slide=state.slides[state.currentSlide];return slide&&slide.objects?slide.objects.find(function(o){return o.id===state.selectedObjectId;})||null:null;}
function selectObject(id){state.selectedObjectId=id;state.selectedTextId=null;if(ui.objectLayerHost)ui.objectLayerHost.querySelectorAll('.object-layer').forEach(function(n){n.classList.toggle('selected',n.dataset.id===id);});if(ui.textLayerHost)ui.textLayerHost.querySelectorAll('.text-layer').forEach(function(n){n.classList.remove('selected');});ui.noSelection.classList.add('hidden');ui.textProperties.classList.add('hidden');ui.objectProperties.classList.remove('hidden');const obj=currentObject();if(!obj)return;ui.objectName.value=obj.name||'Elemento';ui.objectOpacity.value=Math.round((obj.opacity==null?1:obj.opacity)*100);}
function hideObjectProperties(){if(ui.objectProperties)ui.objectProperties.classList.add('hidden');}
