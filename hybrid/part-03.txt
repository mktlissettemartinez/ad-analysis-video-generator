  const padX = Math.max(4, (bbox.y1 - bbox.y0) * 0.18), padY = Math.max(3, (bbox.y1 - bbox.y0) * 0.12); const x = clamp(Math.floor(bbox.x0 - padX), 0, sourceCanvas.width - 1), y = clamp(Math.floor(bbox.y0 - padY), 0, sourceCanvas.height - 1), w = clamp(Math.ceil(bbox.x1 - bbox.x0 + padX * 2), 2, sourceCanvas.width - x), h = clamp(Math.ceil(bbox.y1 - bbox.y0 + padY * 2), 2, sourceCanvas.height - y); const sourceCtx = sourceCanvas.getContext("2d", { willReadFrequently: true });
  const candidates = [regionStats(sourceCtx, x, y - h - 3, w, h), regionStats(sourceCtx, x, y + h + 3, w, h), regionStats(sourceCtx, x - w - 3, y, w, h), regionStats(sourceCtx, x + w + 3, y, w, h)].filter(Boolean).sort((a, b) => a.variance - b.variance);
  targetCtx.save(); targetCtx.beginPath(); targetCtx.rect(x, y, w, h); targetCtx.clip();
  if (candidates.length && candidates[0].variance < 78) { const c = candidates[0]; targetCtx.filter = "blur(1.2px)"; targetCtx.drawImage(sourceCanvas, c.x, c.y, c.w, c.h, x - 2, y - 2, w + 4, h + 4); targetCtx.filter = "none"; }
  else { const borderPoints = []; for (let i = 0; i < 28; i++) { const t = i / 27; borderPoints.push([x + w * t, y - 3], [x + w * t, y + h + 3], [x - 3, y + h * t], [x + w + 3, y + h * t]); } const imageData = sourceCtx.getImageData(0, 0, sourceCanvas.width, sourceCanvas.height); const c = sampleColor(imageData, borderPoints); const gradient = targetCtx.createLinearGradient(x, y, x + w, y + h); gradient.addColorStop(0, `rgb(${clamp255(c[0] * .94)},${clamp255(c[1] * .94)},${clamp255(c[2] * .94)})`); gradient.addColorStop(1, `rgb(${clamp255(c[0] * 1.06)},${clamp255(c[1] * 1.06)},${clamp255(c[2] * 1.06)})`); targetCtx.fillStyle = gradient; targetCtx.fillRect(x, y, w, h); }
  targetCtx.restore();
}
function buildBackground(originalCanvas, lines, themeHex, applyTheme) { const source = document.createElement("canvas"); source.width = originalCanvas.width; source.height = originalCanvas.height; source.getContext("2d").drawImage(originalCanvas, 0, 0); if (applyTheme) recolorCanvas(source, themeHex); const output = document.createElement("canvas"); output.width = source.width; output.height = source.height; const ctx = output.getContext("2d", { willReadFrequently: true }); ctx.drawImage(source, 0, 0); lines.forEach((line) => cleanTextRegion(ctx, source, line.bbox)); return output; }

async function processSlide(source, position, total, settings) {
  state.currentSlide = position; const img = await loadImage(source.blob); const originalCanvas = canvasFromImage(img); const originalDataUrl = originalCanvas.toDataURL("image/jpeg", 0.96); setStage("ocr"); log(`Diapositiva ${position + 1}/${total}: reconociendo textos confiables…`);
  const { canvas: ocrCanvas, scale } = enhancedOcrCanvas(originalCanvas); const worker = await ensureOcrWorker(); const recognition = await worker.recognize(ocrCanvas); const lines = extractLinesFromOcr(recognition.data, scale, originalCanvas.width, originalCanvas.height);
  lines.forEach((line, index) => Object.assign(line, estimateTextStyle(originalCanvas, line), { id: `${source.index}-${index}-${Math.random().toString(36).slice(2, 8)}`, original: JSON.parse(JSON.stringify(line)) }));
  setStage("clean"); const backgroundCanvas = buildBackground(originalCanvas, lines, settings.primaryColor, settings.applyTheme); const backgroundDataUrl = backgroundCanvas.toDataURL("image/jpeg", 0.95); const thumb = document.createElement("canvas"); thumb.width = 320; thumb.height = Math.round(320 * originalCanvas.height / originalCanvas.width); thumb.getContext("2d").drawImage(backgroundCanvas, 0, 0, thumb.width, thumb.height);
  log(`Diapositiva ${position + 1}: ${lines.length} textos convertidos en capas editables.`); return { index: source.index, width: originalCanvas.width, height: originalCanvas.height, originalDataUrl, backgroundDataUrl, thumbnailDataUrl: thumb.toDataURL("image/jpeg", 0.82), lines, originalLines: JSON.parse(JSON.stringify(lines)) };
}

function updateSummary() { ui.slideCount.textContent = state.slides.length; ui.textCount.textContent = state.slides.reduce((sum, s) => sum + s.lines.length, 0); }
function renderThumbs() { ui.thumbs.innerHTML = ""; state.slides.forEach((slide, i) => { const button = document.createElement("button"); button.className = `thumb${i === state.currentSlide ? " active" : ""}`; button.innerHTML = `<img alt="Diapositiva ${slide.index}" src="${slide.thumbnailDataUrl}"><span>${slide.index}</span>`; button.addEventListener("click", () => selectSlide(i)); ui.thumbs.appendChild(button); }); }
function fitStage() { const slide = state.slides[state.currentSlide]; if (!slide) return; const availableW = Math.max(320, ui.slideStageWrap.clientWidth - 26); const availableH = Math.max(260, ui.slideStageWrap.clientHeight - 26); const scale = Math.min(availableW / slide.width, availableH / slide.height, 1); ui.slideStage.style.width = `${slide.width * scale}px`; ui.slideStage.style.height = `${slide.height * scale}px`; ui.slideStage.dataset.scale = String(scale); }
function selectSlide(index) { saveActivePropertyValues(); state.currentSlide = clamp(index, 0, state.slides.length - 1); state.selectedTextId = null; renderThumbs(); renderCurrentSlide(); }
function renderCurrentSlide() { const slide = state.slides[state.currentSlide]; if (!slide) return; ui.slideBackground.src = slide.backgroundDataUrl; fitStage(); ui.textLayerHost.innerHTML = ""; slide.lines.forEach((line) => ui.textLayerHost.appendChild(createTextLayerElement(line, slide))); ui.noSelection.classList.remove("hidden"); ui.textProperties.classList.add("hidden"); }
function createTextLayerElement(line, slide) { const node = document.createElement("div"); node.className = "text-layer"; node.dataset.id = line.id; applyLineToNode(node, line, slide); node.innerHTML = `<div class="drag-handle">MOVER</div><div class="editable" contenteditable="true" spellcheck="true"></div><div class="resize-handle"></div>`; const editable = node.querySelector(".editable"); editable.textContent = line.text; editable.addEventListener("input", () => { line.text = editable.textContent; if (state.selectedTextId === line.id) ui.textContent.value = line.text; }); node.addEventListener("mousedown", (event) => { if (!event.target.classList.contains("drag-handle") && !event.target.classList.contains("resize-handle")) selectText(line.id); }); node.querySelector(".drag-handle").addEventListener("pointerdown", (event) => startDrag(event, line, node, false)); node.querySelector(".resize-handle").addEventListener("pointerdown", (event) => startDrag(event, line, node, true)); return node; }
function applyLineToNode(node, line, slide) { node.style.left = `${(line.bbox.x0 / slide.width) * 100}%`; node.style.top = `${(line.bbox.y0 / slide.height) * 100}%`; node.style.width = `${((line.bbox.x1 - line.bbox.x0) / slide.width) * 100}%`; node.style.height = `${Math.max(1.8, ((line.bbox.y1 - line.bbox.y0) / slide.height) * 100 * 1.35)}%`; node.style.fontFamily = `${ui.fontFace.value || "Poppins"}, sans-serif`; const stageScale = Number(ui.slideStage.dataset.scale || 1); node.style.fontSize = `${line.fontSize * 96 / 72 * stageScale}px`; node.style.color = line.color; node.style.fontWeight = line.bold ? "700" : "400"; node.style.textAlign = line.align; }
function selectText(id) { state.selectedTextId = id; ui.textLayerHost.querySelectorAll(".text-layer").forEach((n) => n.classList.toggle("selected", n.dataset.id === id)); const line = currentLine(); if (!line) return; ui.noSelection.classList.add("hidden"); ui.textProperties.classList.remove("hidden"); ui.textContent.value = line.text; ui.textSize.value = line.fontSize.toFixed(1); ui.textColor.value = line.color; ui.textAlign.value = line.align; ui.textBold.checked = line.bold; }
function currentLine() { return state.slides[state.currentSlide]?.lines.find((l) => l.id === state.selectedTextId) || null; }
function saveActivePropertyValues() { const line = currentLine(); if (!line || ui.textProperties.classList.contains("hidden")) return; line.text = ui.textContent.value; line.fontSize = clamp(Number(ui.textSize.value) || line.fontSize, 6, 72); line.color = ui.textColor.value; line.align = ui.textAlign.value; line.bold = ui.textBold.checked; }
function syncSelectedNode() { const line = currentLine(); if (!line) return; const node = ui.textLayerHost.querySelector(`[data-id="${CSS.escape(line.id)}"]`); if (!node) return; node.querySelector(".editable").textContent = line.text; applyLineToNode(node, line, state.slides[state.currentSlide]); updateSummary(); }
function startDrag(event, line, node, resize) { event.preventDefault(); event.stopPropagation(); selectText(line.id); const slide = state.slides[state.currentSlide]; const scale = Number(ui.slideStage.dataset.scale || 1); const startX = event.clientX, startY = event.clientY; const original = { ...line.bbox }; const onMove = (e) => { const dx = (e.clientX - startX) / scale, dy = (e.clientY - startY) / scale; if (resize) { line.bbox.x1 = clamp(original.x1 + dx, line.bbox.x0 + 20, slide.width); line.bbox.y1 = clamp(original.y1 + dy, line.bbox.y0 + 12, slide.height); } else { const w = original.x1 - original.x0, h = original.y1 - original.y0; line.bbox.x0 = clamp(original.x0 + dx, 0, slide.width - w); line.bbox.y0 = clamp(original.y0 + dy, 0, slide.height - h); line.bbox.x1 = line.bbox.x0 + w; line.bbox.y1 = line.bbox.y0 + h; } applyLineToNode(node, line, slide); }; const onUp = () => { window.removeEventListener("pointermove", onMove); window.removeEventListener("pointerup", onUp); }; window.addEventListener("pointermove", onMove); window.addEventListener("pointerup", onUp); }
