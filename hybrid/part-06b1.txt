function cleanTextRegion(targetCtx,sourceCanvas,bbox){const sourceCtx=sourceCanvas.getContext('2d',{willReadFrequently:true}),W=sourceCanvas.width,H=sourceCanvas.height,h=Math.max(4,bbox.y1-bbox.y0),padX=Math.max(2,h*.08),padY=Math.max(2,h*.06),x0=clamp(Math.floor(bbox.x0-padX),0,W-1),y0=clamp(Math.floor(bbox.y0-padY),0,H-1),x1=clamp(Math.ceil(bbox.x1+padX),1,W),y1=clamp(Math.ceil(bbox.y1+padY),1,H),w=x1-x0,hh=y1-y0;if(w<2||hh<2)return;const full=sourceCtx.getImageData(0,0,W,H),region=targetCtx.getImageData(x0,y0,w,hh),border=[];for(let i=0;i<32;i++){const t=i/31;border.push([x0+w*t,y0-2],[x0+w*t,y1+2],[x0-2,y0+hh*t],[x1+2,y0+hh*t]);}const bg=sampleColor(full,border),bgLum=.2126*bg[0]+.7152*bg[1]+.0722*bg[2],mask=new Uint8Array(w*hh);for(let y=0;y<hh;y++)for(let x=0;x<w;x++){const i=(y*w+x)*4,r=region.data[i],g=region.data[i+1],b=region.data[i+2],lum=.2126*r+.7152*g+.0722*b,dist=Math.abs(lum-bgLum)+(Math.abs(r-bg[0])+Math.abs(g-bg[1])+Math.abs(b-bg[2]))/5;if(dist>52)mask[y*w+x]=1;}const grown=binaryDilate(mask,w,hh,1);for(let y=0;y<hh;y++)for(let x=0;x<w;x++){if(!grown[y*w+x])continue;const i=(y*w+x)*4,tx=x/Math.max(1,w-1),ty=y/Math.max(1,hh-1),shade=(tx+ty-1)*4;region.data[i]=clamp255(bg[0]+shade);region.data[i+1]=clamp255(bg[1]+shade);region.data[i+2]=clamp255(bg[2]+shade);region.data[i+3]=255;}targetCtx.putImageData(region,x0,y0);}
