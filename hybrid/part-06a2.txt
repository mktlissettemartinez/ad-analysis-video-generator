function createObjectLayerElement(obj,slide){const node=document.createElement('div');node.className='object-layer';node.dataset.id=obj.id;node.innerHTML='<img alt="Elemento"><div class="object-drag">MOVER</div><div class="object-resize"></div>';node.querySelector('img').src=obj.dataUrl;applyObjectToNode(node,obj,slide);node.addEventListener('mousedown',function(){selectObject(obj.id);});node.querySelector('.object-drag').addEventListener('pointerdown',function(e){startObjectDrag(e,obj,node,false);});node.querySelector('.object-resize').addEventListener('pointerdown',function(e){startObjectDrag(e,obj,node,true);});return node;}
function applyObjectToNode(node,obj,slide){node.style.left=(obj.bbox.x0/slide.width*100)+'%';node.style.top=(obj.bbox.y0/slide.height*100)+'%';node.style.width=((obj.bbox.x1-obj.bbox.x0)/slide.width*100)+'%';node.style.height=((obj.bbox.y1-obj.bbox.y0)/slide.height*100)+'%';node.style.opacity=String(obj.opacity==null?1:obj.opacity);const img=node.querySelector('img');if(img&&img.src!==obj.dataUrl)img.src=obj.dataUrl;}
function startObjectDrag(event,obj,node,resize){event.preventDefault();event.stopPropagation();selectObject(obj.id);const slide=state.slides[state.currentSlide],scale=Number(ui.slideStage.dataset.scale||1),sx=event.clientX,sy=event.clientY,original=Object.assign({},obj.bbox);function move(e){const dx=(e.clientX-sx)/scale,dy=(e.clientY-sy)/scale;if(resize){obj.bbox.x1=clamp(original.x1+dx,obj.bbox.x0+20,slide.width);obj.bbox.y1=clamp(original.y1+dy,obj.bbox.y0+20,slide.height);}else{const w=original.x1-original.x0,h=original.y1-original.y0;obj.bbox.x0=clamp(original.x0+dx,0,slide.width-w);obj.bbox.y0=clamp(original.y0+dy,0,slide.height-h);obj.bbox.x1=obj.bbox.x0+w;obj.bbox.y1=obj.bbox.y0+h;}applyObjectToNode(node,obj,slide);}function up(){window.removeEventListener('pointermove',move);window.removeEventListener('pointerup',up);}window.addEventListener('pointermove',move);window.addEventListener('pointerup',up);}
