function addText() { const slide = state.slides[state.currentSlide]; if (!slide) return; const line = { id: `${slide.index}-manual-${Date.now()}`, text: "Nuevo texto", confidence: 100, bbox: { x0: slide.width * .25, y0: slide.height * .43, x1: slide.width * .75, y1: slide.height * .52 }, color: "#FFFFFF", fontSize: 24, bold: false, align: "center", original: null }; slide.lines.push(line); renderCurrentSlide(); selectText(line.id); updateSummary(); }
function deleteSelectedText() { const slide = state.slides[state.currentSlide]; if (!slide || !state.selectedTextId) return; slide.lines = slide.lines.filter((l) => l.id !== state.selectedTextId); state.selectedTextId = null; renderCurrentSlide(); updateSummary(); }
function resetCurrentSlide() { const slide = state.slides[state.currentSlide]; if (!slide) return; if (!confirm("¿Restaurar los textos detectados originalmente en esta diapositiva?")) return; slide.lines = JSON.parse(JSON.stringify(slide.originalLines)); renderCurrentSlide(); updateSummary(); }

async function generatePowerPoint() {
  saveActivePropertyValues(); setStatus("Generando", "working"); ui.panelTitle.textContent = "Generando PowerPoint editable"; const pptx = new PptxGenJS(); pptx.author = "Editor híbrido de diapositivas"; pptx.company = "Editor híbrido de diapositivas"; pptx.subject = "Presentación híbrida editable"; pptx.title = sanitizeFilename(state.file.name); pptx.lang = "es-SV"; pptx.defineLayout({ name: "HYBRID", width: state.parsed.widthIn, height: state.parsed.heightIn }); pptx.layout = "HYBRID"; pptx.theme = { headFontFace: ui.fontFace.value || "Poppins", bodyFontFace: ui.fontFace.value || "Poppins", lang: "es-SV" };
  for (let i = 0; i < state.slides.length; i++) { const source = state.slides[i]; const slide = pptx.addSlide(); slide.addImage({ data: source.backgroundDataUrl, x: 0, y: 0, w: state.parsed.widthIn, h: state.parsed.heightIn }); for (const line of source.lines) { const x = line.bbox.x0 / source.width * state.parsed.widthIn, y = line.bbox.y0 / source.height * state.parsed.heightIn, w = Math.max(.1, (line.bbox.x1 - line.bbox.x0) / source.width * state.parsed.widthIn), h = Math.max(.1, (line.bbox.y1 - line.bbox.y0) / source.height * state.parsed.heightIn * 1.35); slide.addText(line.text, { x, y, w, h, fontFace: ui.fontFace.value || "Poppins", fontSize: line.fontSize, color: line.color.replace("#", ""), bold: line.bold, align: line.align, valign: "mid", margin: 0, breakLine: false, fit: "shrink", autoFit: false, transparency: 0 }); } setProgress(88 + ((i + 1) / state.slides.length) * 11, `Generando diapositiva ${i + 1} de ${state.slides.length}`); await new Promise((r) => setTimeout(r, 0)); }
  state.outputBlob = await pptx.write({ outputType: "blob" }); const filename = `${sanitizeFilename(state.file.name)}_hibrido_editable.pptx`; downloadBlob(state.outputBlob, filename); ui.downloadBtn.classList.remove("hidden"); setProgress(100, "Completado"); setStatus("Completado", "done"); ui.panelTitle.textContent = "Presentación editable lista"; log(`Archivo generado: ${filename} (${formatBytes(state.outputBlob.size)})`);
}

async function startProcessing() {
  if (!state.file || state.running) return; const primary = ui.primaryColor.value.trim(); if (ui.applyTheme.checked && !/^#[0-9a-f]{6}$/i.test(primary)) { alert("Escribe el color principal en formato hexadecimal, por ejemplo #1c21d1"); return; }
  state.running = true; ui.processBtn.disabled = true; ui.removeFileBtn.disabled = true; ui.emptyState.classList.add("hidden"); ui.editorView.classList.add("hidden"); ui.processingView.classList.remove("hidden"); ui.downloadBtn.classList.add("hidden"); ui.logBox.textContent = ""; setStatus("Procesando", "working"); ui.panelTitle.textContent = "Preparando la presentación"; setProgress(1, "Comenzando…"); STAGES.forEach((s) => ui.stageGrid.querySelector(`[data-stage="${s}"]`)?.classList.remove("active", "complete"));
  try { state.parsed = await parsePresentation(state.file); state.slides = []; const settings = { primaryColor: primary, applyTheme: ui.applyTheme.checked }; for (let i = 0; i < state.parsed.slides.length; i++) { state.slides.push(await processSlide(state.parsed.slides[i], i, state.parsed.slides.length, settings)); setProgress(18 + ((i + 1) / state.parsed.slides.length) * 68, `Diapositiva ${i + 1} de ${state.parsed.slides.length} completada`); await new Promise((r) => setTimeout(r, 0)); } setStage("review"); setProgress(88, "Abriendo editor…"); state.currentSlide = 0; ui.processingView.classList.add("hidden"); ui.editorView.classList.remove("hidden"); ui.panelTitle.textContent = "Revisa y corrige antes de descargar"; setStatus("Lista para revisar", "done"); updateSummary(); renderThumbs(); renderCurrentSlide(); ui.downloadBtn.classList.remove("hidden"); setProgress(100, "Listo para revisar"); }
  catch (error) { console.error(error); log(`ERROR: ${error.message || error}`); setStatus("Error", "error"); ui.panelTitle.textContent = "No se pudo procesar el archivo"; alert(`No se pudo completar el proceso.\n\n${error.message || error}`); }
  finally { state.running = false; ui.processBtn.disabled = false; ui.removeFileBtn.disabled = false; }
}

ui.textContent.addEventListener("input", () => { const line = currentLine(); if (line) { line.text = ui.textContent.value; syncSelectedNode(); } }); ui.textSize.addEventListener("input", () => { const line = currentLine(); if (line) { line.fontSize = clamp(Number(ui.textSize.value) || line.fontSize, 6, 72); syncSelectedNode(); } }); ui.textColor.addEventListener("input", () => { const line = currentLine(); if (line) { line.color = ui.textColor.value; syncSelectedNode(); } }); ui.textAlign.addEventListener("change", () => { const line = currentLine(); if (line) { line.align = ui.textAlign.value; syncSelectedNode(); } }); ui.textBold.addEventListener("change", () => { const line = currentLine(); if (line) { line.bold = ui.textBold.checked; syncSelectedNode(); } });
window.addEventListener("resize", () => { if (!ui.editorView.classList.contains("hidden")) { fitStage(); renderCurrentSlide(); } }); window.addEventListener("beforeunload", async () => { try { if (state.worker) await state.worker.terminate(); } catch (_) {} });
