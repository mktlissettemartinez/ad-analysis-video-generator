function ensureSlideObjects(slide) {
  if (!slide.objects) slide.objects = [];
  if (!slide.baseBackgroundDataUrl) slide.baseBackgroundDataUrl = slide.backgroundDataUrl;
  if (!slide.originalObjects) slide.originalObjects = [];
}

function setSelectionBrush(erase) { state.selection.erase = erase; ui.paintSelectBtn.classList.toggle("active", !erase); ui.eraseSelectBtn.classList.toggle("active", erase); }
function createMaskCanvas(w, h) { const c = document.createElement("canvas"); c.width = w; c.height = h; return c; }
function startSelectionMode() {
  const slide = state.slides[state.currentSlide]; if (!slide) return; ensureSlideObjects(slide); state.selectedTextId = null; state.selectedObjectId = null;
  state.selection.active = true; state.selection.erase = false; state.selection.brush = Number(ui.brushSize.value); state.selection.tolerance = Number(ui.selectionTolerance.value);
  state.selection.positive = createMaskCanvas(slide.width, slide.height); state.selection.negative = createMaskCanvas(slide.width, slide.height);
  ui.selectionCanvas.width = slide.width; ui.selectionCanvas.height = slide.height; ui.selectionCanvas.classList.add("active"); ui.selectionToolbar.classList.remove("hidden"); ui.isolateElementBtn.classList.add("active"); setSelectionBrush(false); renderSelectionOverlay();
}
function cancelSelectionMode() { state.selection.active = false; state.selection.drawing = false; ui.selectionCanvas.classList.remove("active"); ui.selectionToolbar.classList.add("hidden"); ui.isolateElementBtn.classList.remove("active"); const c = ui.selectionCanvas.getContext("2d"); c.clearRect(0,0,ui.selectionCanvas.width,ui.selectionCanvas.height); }
function pointerToSlide(event) { const r = ui.selectionCanvas.getBoundingClientRect(); return { x:(event.clientX-r.left)/r.width*ui.selectionCanvas.width, y:(event.clientY-r.top)/r.height*ui.selectionCanvas.height }; }
function paintMaskPoint(x,y) {
  const radius = state.selection.brush / Math.max(0.25, Number(ui.slideStage.dataset.scale || 1)) / 2; const target = state.selection.erase ? state.selection.negative : state.selection.positive; const other = state.selection.erase ? state.selection.positive : state.selection.negative;
  const t = target.getContext("2d"), o = other.getContext("2d"); t.fillStyle="#fff"; t.beginPath(); t.arc(x,y,radius,0,Math.PI*2); t.fill(); o.globalCompositeOperation="destination-out"; o.beginPath(); o.arc(x,y,radius,0,Math.PI*2); o.fill(); o.globalCompositeOperation="source-over"; renderSelectionOverlay();
}
function selectionPointerDown(e){ if(!state.selection.active)return; e.preventDefault(); ui.selectionCanvas.setPointerCapture?.(e.pointerId); state.selection.drawing=true; const p=pointerToSlide(e); paintMaskPoint(p.x,p.y); }
function selectionPointerMove(e){ if(!state.selection.active||!state.selection.drawing)return; e.preventDefault(); const p=pointerToSlide(e); paintMaskPoint(p.x,p.y); }
function selectionPointerUp(){ state.selection.drawing=false; }
function renderSelectionOverlay(){ const ctx=ui.selectionCanvas.getContext("2d"); ctx.clearRect(0,0,ui.selectionCanvas.width,ui.selectionCanvas.height); if(!state.selection.active)return; ctx.save(); ctx.globalAlpha=.45; ctx.drawImage(state.selection.positive,0,0); ctx.globalCompositeOperation="source-in"; ctx.fillStyle="#00d7ff"; ctx.fillRect(0,0,ui.selectionCanvas.width,ui.selectionCanvas.height); ctx.restore(); ctx.save(); ctx.globalAlpha=.42; ctx.drawImage(state.selection.negative,0,0); ctx.globalCompositeOperation="source-in"; ctx.fillStyle="#ff416c"; ctx.fillRect(0,0,ui.selectionCanvas.width,ui.selectionCanvas.height); ctx.restore(); }

function resizeMaskCanvas(source,w,h){ const c=createMaskCanvas(w,h); const ctx=c.getContext("2d"); ctx.imageSmoothingEnabled=false; ctx.drawImage(source,0,0,w,h); return c; }
function binaryDilate(mask,w,h,passes=1){ let src=mask; for(let p=0;p<passes;p++){ const out=new Uint8Array(src.length); for(let y=1;y<h-1;y++)for(let x=1;x<w-1;x++){ let v=0; for(let yy=-1;yy<=1&&!v;yy++)for(let xx=-1;xx<=1;xx++)if(src[(y+yy)*w+x+xx]){v=1;break;} out[y*w+x]=v;} src=out;} return src; }
function binaryErode(mask,w,h,passes=1){ let src=mask; for(let p=0;p<passes;p++){ const out=new Uint8Array(src.length); for(let y=1;y<h-1;y++)for(let x=1;x<w-1;x++){ let v=1; for(let yy=-1;yy<=1&&v;yy++)for(let xx=-1;xx<=1;xx++)if(!src[(y+yy)*w+x+xx]){v=0;break;} out[y*w+x]=v;} src=out;} return src; }
function buildConnectedSelection(imageCanvas,posCanvas,negCanvas,tolerance){
  const maxW=820, scale=Math.min(1,maxW/imageCanvas.width), w=Math.max(1,Math.round(imageCanvas.width*scale)), h=Math.max(1,Math.round(imageCanvas.height*scale));
  const imgC=createMaskCanvas(w,h); imgC.getContext("2d").drawImage(imageCanvas,0,0,w,h); const data=imgC.getContext("2d",{willReadFrequently:true}).getImageData(0,0,w,h).data;
  const pos=resizeMaskCanvas(posCanvas,w,h).getContext("2d").getImageData(0,0,w,h).data, neg=resizeMaskCanvas(negCanvas,w,h).getContext("2d").getImageData(0,0,w,h).data;
  const seeds=[], palette=[]; for(let i=0;i<w*h;i++){ if(pos[i*4+3]>20){ seeds.push(i); if(palette.length<80||i%(Math.max(1,Math.floor(seeds.length/80)))===0) palette.push([data[i*4],data[i*4+1],data[i*4+2]]); } }
  if(!seeds.length) throw new Error("Pinta sobre el elemento antes de crear la capa.");
  while(palette.length>64) palette.splice(Math.floor(Math.random()*palette.length),1);
  const candidate=new Uint8Array(w*h), threshold=Math.max(18,tolerance)*2.15;
  for(let i=0;i<w*h;i++){ if(neg[i*4+3]>20)continue; const r=data[i*4],g=data[i*4+1],b=data[i*4+2]; let best=1e9; for(const c of palette){ const d=Math.sqrt((r-c[0])**2+(g-c[1])**2+(b-c[2])**2); if(d<best)best=d; if(best<=threshold)break; } if(best<=threshold||pos[i*4+3]>20)candidate[i]=1; }
  const connected=new Uint8Array(w*h), queue=new Int32Array(w*h); let q0=0,q1=0; for(const s of seeds){ if(candidate[s]&&!connected[s]){connected[s]=1;queue[q1++]=s;} }
  while(q0<q1){ const idx=queue[q0++],x=idx%w,y=(idx/w)|0; for(let yy=-1;yy<=1;yy++)for(let xx=-1;xx<=1;xx++){if(!xx&&!yy)continue;const nx=x+xx,ny=y+yy;if(nx<0||ny<0||nx>=w||ny>=h)continue;const ni=ny*w+nx;if(candidate[ni]&&!connected[ni]){connected[ni]=1;queue[q1++]=ni;}} }
  let mask=binaryDilate(connected,w,h,1); mask=binaryErode(mask,w,h,1); mask=binaryDilate(mask,w,h,1);
  let minX=w,minY=h,maxX=-1,maxY=-1,count=0; for(let i=0;i<mask.length;i++)if(mask[i]){const x=i%w,y=(i/w)|0;minX=Math.min(minX,x);minY=Math.min(minY,y);maxX=Math.max(maxX,x);maxY=Math.max(maxY,y);count++;}
  if(count<40||maxX<minX) throw new Error("No se pudo aislar un área suficiente. Pinta más partes del elemento o aumenta la precisión.");
  const maskCanvas=createMaskCanvas(w,h), md=maskCanvas.getContext("2d").createImageData(w,h); for(let i=0;i<mask.length;i++)if(mask[i])md.data[i*4+3]=255; maskCanvas.getContext("2d").putImageData(md,0,0);
  return {maskCanvas,scale,w,h,bboxSmall:{minX,minY,maxX,maxY},count};
}

async function createLayerFromSelection(){
  const slide=state.slides[state.currentSlide]; if(!slide||!state.selection.active)return;
  try{
    ui.createLayerBtn.disabled=true; ui.createLayerBtn.textContent="Separando…"; const image=await loadImage(slide.backgroundDataUrl); const source=canvasFromImage(image); const built=buildConnectedSelection(source,state.selection.positive,state.selection.negative,state.selection.tolerance);
    const fullMask=createMaskCanvas(slide.width,slide.height), fm=fullMask.getContext("2d"); fm.imageSmoothingEnabled=false; fm.drawImage(built.maskCanvas,0,0,slide.width,slide.height);
    const maskData=fm.getImageData(0,0,slide.width,slide.height).data; let x0=slide.width,y0=slide.height,x1=0,y1=0; for(let y=0;y<slide.height;y+=2)for(let x=0;x<slide.width;x+=2)if(maskData[(y*slide.width+x)*4+3]>20){x0=Math.min(x0,x);y0=Math.min(y0,y);x1=Math.max(x1,x);y1=Math.max(y1,y);} x0=Math.max(0,x0-4);y0=Math.max(0,y0-4);x1=Math.min(slide.width,x1+5);y1=Math.min(slide.height,y1+5);
    const cw=Math.max(2,x1-x0),ch=Math.max(2,y1-y0), objCanvas=createMaskCanvas(cw,ch), oc=objCanvas.getContext("2d"); oc.drawImage(source,x0,y0,cw,ch,0,0,cw,ch); oc.globalCompositeOperation="destination-in"; oc.filter="blur(1px)"; oc.drawImage(fullMask,x0,y0,cw,ch,0,0,cw,ch); oc.filter="none"; oc.globalCompositeOperation="source-over";
    const objectDataUrl=objCanvas.toDataURL("image/png"); const newBackground=fillSelectionHole(source,fullMask,{x0,y0,x1,y1}); slide.backgroundDataUrl=newBackground.toDataURL("image/jpeg",.96); const thumb=createMaskCanvas(320,Math.round(320*slide.height/slide.width)); thumb.getContext("2d").drawImage(newBackground,0,0,thumb.width,thumb.height); slide.thumbnailDataUrl=thumb.toDataURL("image/jpeg",.82);
    const obj={id:`${slide.index}-object-${Date.now()}`,name:`Elemento ${slide.objects.length+1}`,bbox:{x0,y0,x1,y1},dataUrl:objectDataUrl,originalDataUrl:objectDataUrl,opacity:1}; slide.objects.push(obj); cancelSelectionMode(); renderThumbs(); renderCurrentSlide(); selectObject(obj.id); updateSummary();
  }catch(error){alert(error.message||String(error));}finally{ui.createLayerBtn.disabled=false;ui.createLayerBtn.textContent="Crear capa";}
}

function fillSelectionHole(source,maskCanvas,bbox){
  const out=createMaskCanvas(source.width,source.height),ctx=out.getContext("2d",{willReadFrequently:true});ctx.drawImage(source,0,0); const img=ctx.getImageData(0,0,out.width,out.height),mask=maskCanvas.getContext("2d").getImageData(0,0,out.width,out.height).data;
  const pad=16,x0=Math.max(0,bbox.x0-pad),y0=Math.max(0,bbox.y0-pad),x1=Math.min(out.width,bbox.x1+pad),y1=Math.min(out.height,bbox.y1+pad);
  for(let y=y0;y<y1;y++){ let left=-1,right=-1; for(let x=bbox.x0-1;x>=x0;x--)if(mask[(y*out.width+x)*4+3]<20){left=x;break;} for(let x=bbox.x1;x<x1;x++)if(mask[(y*out.width+x)*4+3]<20){right=x;break;} for(let x=bbox.x0;x<bbox.x1;x++){const mi=(y*out.width+x)*4;if(mask[mi+3]<20)continue;let srcX;if(left>=0&&right>=0){const t=(x-bbox.x0)/Math.max(1,bbox.x1-bbox.x0);const li=(y*out.width+left)*4,ri=(y*out.width+right)*4;img.data[mi]=img.data[li]*(1-t)+img.data[ri]*t;img.data[mi+1]=img.data[li+1]*(1-t)+img.data[ri+1]*t;img.data[mi+2]=img.data[li+2]*(1-t)+img.data[ri+2]*t;img.data[mi+3]=255;}else{srcX=left>=0?left:right;if(srcX>=0){const si=(y*out.width+srcX)*4;img.data[mi]=img.data[si];img.data[mi+1]=img.data[si+1];img.data[mi+2]=img.data[si+2];img.data[mi+3]=255;}}}}
  ctx.putImageData(img,0,0); const blur=createMaskCanvas(out.width,out.height),bc=blur.getContext("2d");bc.filter="blur(4px)";bc.drawImage(out,0,0);bc.filter="none";ctx.save();ctx.globalCompositeOperation="source-over";ctx.drawImage(blur,bbox.x0,bbox.y0,bbox.x1-bbox.x0,bbox.y1-bbox.y0,bbox.x0,bbox.y0,bbox.x1-bbox.x0,bbox.y1-bbox.y0);ctx.restore();return out;
}
